Linear Classifier¶

A linear classifier is a machine learning model that makes predictions based on a linear combination of input features. It essentially divides the input space into classes by finding a decision boundary (such as a line, plane, or hyperplane, depending on the input dimensionality) that separates instances of different classes. The model uses this decision boundary to classify new instances by determining on which side of the boundary they fall. Linear classifiers are simple yet powerful tools for classification tasks, often serving as a baseline for more complex algorithms. They are particularly well-suited for problems where the classes can be separated by a linear boundary.

def calculate_loss(y_pred, y_true):
    # Assuming y_true is not one-hot encoded and y_pred is softmax probabilities
    m = y_true.shape[0]
    log_probs = -np.log(y_pred[range(m), y_true])
    loss = np.sum(log_probs) / m
    return loss


# Losses: Softmax, cross-entropy loss
def softmax(x):
    exp_x = np.exp(x - np.max(x, axis=1, keepdims=True))
    return exp_x / np.sum(exp_x, axis=1, keepdims=True)
def cross_entropy_loss(y_pred, y_true):
    m = y_true.shape[0]
    return -np.sum(y_true * np.log(y_pred)) / m


#Accuracy
def calculate_accuracy(X, y, W, b):
    # Compute the scores
    scores = np.dot(X, W) + b
    # Apply softmax to get probabilities
    probabilities = softmax(scores)
    # Get predicted labels by finding the index of the max probability
    predictions = np.argmax(probabilities, axis=1)   
    # If y is one-hot encoded, convert it back to label indices for comparison
    if y.ndim > 1 and y.shape[1] > 1:
        y_true = np.argmax(y, axis=1)
    else:
        y_true = y    
    # Calculate accuracy
    accuracy = np.mean(predictions == y_true)
    return accuracy

def display_random_images(X, y, num_images=10):
    # Select random indices
    random_indices = np.random.choice(X.shape[0], num_images, replace=False)
    images = X[random_indices]
    labels = y[random_indices]

    # Plot images in a 2x5 grid
    fig, axes = plt.subplots(nrows=2, ncols=5, figsize=(15, 6))
    for i, ax in enumerate(axes.flat):
        ax.imshow(images[i].reshape(28, 28), cmap='gray')
        ax.set_title(f'Label: {labels[i]}')
        ax.axis('off')
    plt.tight_layout()
    plt.show()
def visualize_predictions(X, y, W, b, num_images=10):
    # Make predictions
    scores = np.dot(X, W) + b
    y_pred_prob = softmax(scores)
    y_pred = np.argmax(y_pred_prob, axis=1)

    # Convert y to indices if it's one-hot encoded
    if y.ndim > 1 and y.shape[1] > 1:
        y_true = np.argmax(y, axis=1)
    else:
        y_true = y
    # Select random images
    random_indices = np.random.choice(X.shape[0], num_images, replace=False)
    selected_images = X[random_indices]
    selected_true_labels = y_true[random_indices]
    selected_pred_labels = y_pred[random_indices]
    selected_pred_probs = y_pred_prob[random_indices, selected_pred_labels]
    # Calculate loss for each selected image
    losses = [calculate_loss(y_pred_prob[i].reshape(1, -1), np.array([y_true[i]])) for i in random_indices]
    # Visualization
    fig, axes = plt.subplots(nrows=2, ncols=5, figsize=(15, 6))
    for i, ax in enumerate(axes.flat):
        idx = random_indices[i]
        image = selected_images[i].reshape(28, 28)

        pred_label = selected_pred_labels[i]

        ax.set_title(f'True: {true_label}, Pred: {pred_label}\nLoss: {loss:.4f}')
        ax.axis('off')
    plt.tight_layout()
    plt.show()
